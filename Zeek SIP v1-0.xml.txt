<form version="1.1" theme="dark">
  <label>Zeek SIP (V1.0)</label>
  <description>V1 Additions: First generation of dashboard</description>
  <fieldset submitButton="true" autoRun="true">
    <input type="time" token="time">
      <label>Time Selection</label>
      <default>
        <earliest>-24h@h</earliest>
        <latest>now</latest>
      </default>
    </input>
  </fieldset>
  <row>
    <panel>
      <single>
        <title>Total SIP Logs</title>
        <search>
          <query>index=<index> sourcetype="zeek:sip:json"
| stats count</query>
          <earliest>$time.earliest$</earliest>
          <latest>$time.latest$</latest>
          <refresh>5m</refresh>
          <refreshType>delay</refreshType>
        </search>
        <option name="drilldown">none</option>
        <option name="refresh.display">progressbar</option>
      </single>
    </panel>
  </row>
  <row>
    <panel>
      <table>
        <title>SIP Requests</title>
        <search>
          <query>index=<index> sourcetype="zeek:sip:json"   NOT 
    [| inputlookup Known_Scanners.csv
    | rename scannerip AS id.orig_h
    | table id.orig_h
    | format] 
| fillnull value="-" uri, user_agent, id.orig_h, request_from, id.resp_h
| stats count by uri, user_agent, seq, id.orig_h,  id.resp_h, request_from
| where !like (seq, "63104 OPTIONS")
| sort -count
| rename uri AS URI, user_agent AS "User Agent", seq AS "Request Type", id.orig_h AS Source, id.resp_h AS Destination, request_from AS "Destination Info"</query>
          <earliest>$time.earliest$</earliest>
          <latest>$time.latest$</latest>
          <refresh>5m</refresh>
          <refreshType>delay</refreshType>
        </search>
        <option name="drilldown">none</option>
        <option name="refresh.display">progressbar</option>
      </table>
    </panel>
  </row>
  <row>
    <panel>
      <table>
        <title>SIP Responses</title>
        <search>
          <query>index=<index> sourcetype="zeek:sip:json"   NOT 
    [| inputlookup Known_Scanners.csv
    | rename scannerip AS id.orig_h
    | table id.orig_h
    | format] 
| fillnull value="-" warning
| stats count by status_code, status_msg,warning, id.orig_h, id.resp_h, response_from
| sort -count
| rename status_code AS "Status Code",status_msg AS Status, warning AS Message, id.orig_h AS Source, id.resp_h AS Destination, response_from AS "Destination Info", count AS Count</query>
          <earliest>$time.earliest$</earliest>
          <latest>$time.latest$</latest>
          <refresh>5m</refresh>
          <refreshType>delay</refreshType>
        </search>
        <option name="drilldown">none</option>
        <option name="refresh.display">progressbar</option>
      </table>
    </panel>
  </row>
</form>
